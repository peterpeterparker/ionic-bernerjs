import { AlertController } from 'ionic-angular';

@IonicPage()
@Component({
    templateUrl: 'demo.html',
    selector: 'demo'
})
export class Demo {
	constructor(private alertCtrl: AlertController) {
	}

	ionViewDidEnter() {
		this.presentAlert();
	}

	presentAlert() {
		let alert = this.alertCtrl.create({
			title: 'DEMO',
			subTitle: 'Cool?',
			buttons: ['Cool cool cool']
		});

		alert.present();
	}
}
